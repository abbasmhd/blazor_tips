How to read Window Dimensions in Blazor?

Mo/2019-10-02

Blazor.tips How to read the browser window dimensions

window.innerHeight and window.innerWidth

first create a service class to read the dimensions from javascript

1using Microsoft.JSInterop;
2using System.Threading.Tasks;
3
4public class BrowserService
5{
6  private readonly IJSRuntime _js;
7
8  public BrowserService(IJSRuntime js)
9  {
10    _js = js;
11  }
12
13  public async Task<BrowserDimension> GetDimensions()
14  {
15    return await _js.InvokeAsync<BrowserDimension>("getDimensions");
16  }
17}
18
19public class BrowserDimension
20{
21  public int Width { get; set; }
22  public int Height { get; set; }
23}

register the service with the DI on startup.cs

1public void ConfigureServices(IServiceCollection services)
2{
3    services.AddRazorPages();
4    services.AddServerSideBlazor();
5    services.AddScoped<BrowserService>(); // scoped service
6}

add the javascript codes to read the window dimensions you can add them to the _Host.cshtml file directly or add them to a js file 

1    <script type="text/javascript">
2window.getDimensions = function() {
3    return {
4            width: window.innerWidth,
5            height: window.innerHeight
6        };
7};
8</script>

add new razor file to display the dimensions

1@page "/dimensions"
2
3@using BlazorApp.Service
4@inject BrowserService Service
5
6<h1>Window Dimensions</h1>
7
8<p>Window Height: @Height</p>
9<p>Window Width: @Width</p>
10
11<button @onclick="GetDimensions">Get Dimensions</button>
12
13@code {
14
15    public int Height { get; set; }
16    public int Width { get; set; }
17
18    async Task GetDimensions() {
19        var dimension = await Service.GetDimensions();
20        Height = dimension.Height;
21        Width = dimension.Width;
22    }
23
24}
